Snare - rullante
const lowPass = new Tone.Filter({ frequency: 8000 });
const snare = new Tone.NoiseSynth({
  volume: 3,
  noise: {
    type: "pink",
    playbackRate: 3,
  },
  envelope: {
    attack: 0.001,
    decay: 0.2,
    sustain: 0.15,
    release: 0.03,
  },
});

snare.chain(lowPass, Tone.Destination);

snare.triggerAttackRelease("8n");
