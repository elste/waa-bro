Hello Web Audio API
AudioContext
let context;
function init() {
    try {
      context = new AudioContext();
    }
    catch(e) {
      alert('Web Audio API is not supported');
    }
}
Oscillator
let osc = context.createOscillator();
osc.connect(context.destination);
osc.start(context.currentTime);
osc.stop(context.currentTime + 3);

osc.type = "triangle";
